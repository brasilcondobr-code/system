
##  Iniciar o Projeto
    $ cd /Projects/brasilcondo/laravel
    $ ./vendor/bin/sail up -d


##  Limpar cache do laravel
    $ ./vendor/bin/sail artisan optimize:clear


##  Iniciar o SERVIDOR de DESENVOLVIMENTO do laravel
    $ ./vendor/bin/sail npm run dev


##  Acesso pelo Navegador
    URL:                http://localhost/admin/login
    Usuário/Email:      brasilcondo.br@gmail.com
    Senha:              qwerty2099


##  Acessar o Log do Laravel
    $ ./vendor/bin/sail shell
    $ tail -f storage/logs/laravel.log


##  Backup do Projeto  (brasilcondo)
    $ cd ~/Projects/brasilcondo
    $ ./vendor/bin/sail up -d
    $ ./backup_laravel.sh
    $ ./vendor/bin/sail down
    $ ./vendor/bin/sail artisan optimize:clear
    $ ./vendor/bin/sail npm run dev


##  Restauração do Projeto (brasilcondo)
    $ cd ~/Projects/brasilcondo
    $ ./vendor/bin/sail up -d
    $ ./restore_laravel.sh    
    $ ./vendor/bin/sail artisan optimize:clear
    $ ./vendor/bin/sail npm run dev
